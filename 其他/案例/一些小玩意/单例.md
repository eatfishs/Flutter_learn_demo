class Singleton {
  Singleton._();

  static final Singleton _instance = Singleton._();

  static Singleton get instance => _instance;

  factory Singleton() => _instance;

  void init() {
    print('做相应初始化');
  }

  int get age => 10+1;

}


  Singleton.instance.init();
  print(Singleton.instance.age);
