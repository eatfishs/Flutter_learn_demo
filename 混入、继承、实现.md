> [ Dart中抽象类abstract、接口implements、混入Mixin ]( https://juejin.cn/post/6946158943240454181  )<br/>

```

// 普通类
class A {
  var a = 0;
  printA() {
    print('a');
  }
}

// 混入类
mixin B {
  var b = 0;
  printB() {
    print('b');
  }
}

// 抽象类
abstract class C {
  var c = 0;
  printC();

  /// 抽象方法
  printCC() {
    print('我是抽象类中的非抽象方法');
  }
}

/// 继承
class One extends A {
  onePrint() {
    print(super.a);
    print(super.printA());
  }
}

// class ONe extends B{
//   // error:
//   // Classes can only extend other classes.
//   // Try specifying a different superclass, or removing the extends clause.dartextends_non_class
// }
class ONE extends C {
  @override
  printC() {
    // TODO: implement printC
    throw UnimplementedError();
  }

  onePrint() {
    print(super.c);
    print(super.printCC());
  }
}

/// 实现
class Two implements A {
  @override
  late int a;
  @override
  printA() {
    // TODO: implement printA
    throw UnimplementedError();
  }
}

class TWo implements B {
  @override
  late int b;
  @override
  printB() {
    // TODO: implement printB
    throw UnimplementedError();
  }
}

class TWO implements C {
  @override
  late int c;

  @override
  printC() {
    // TODO: implement printC
    throw UnimplementedError();
  }

  @override
  printCC() {
    // TODO: implement printCC
    throw UnimplementedError();
  }
}

/// with 混入 类似于多继承
class Three with A {
  printThree() {
    print(super.a);
    print(super.printA());
  }
}

class THree with B {
  printThree() {
    print(super.b);
    print(super.printB());
  }
}

class THRee with C {
  @override
  printC() {
    // TODO: implement printC
    throw UnimplementedError();
  }

  printThree() {
    print(super.c);
    print(super.printCC());
    // print(super.printC());  // error: The method 'printC' is always abstract in the supertype.dartabstract_super_member_reference
  }
}

class THREe with A, B, C {
  @override
  printC() {
    // TODO: implement printC
    throw UnimplementedError();
  }

  printThree() {
    print(super.a);
    print(super.printA());

    print(super.b);
    print(super.printB());

    print(super.c);
    print(super.printCC());
    // print(super.printC());  // error: The method 'printC' is always abstract in the supertype.dartabstract_super_member_reference
  }
}

```
